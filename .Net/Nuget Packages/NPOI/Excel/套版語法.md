# 套版語法

```cs
private XSSFWorkbook GetWorkbookFromFile()
{
    var       templateFile       = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Templates", "MoatStatusTemplate.xlsx");
    using var templateFileStream = new FileStream(templateFile, FileMode.Open, FileAccess.Read);
    var       workbook           = new XSSFWorkbook(templateFileStream);
    return workbook;
}
```
