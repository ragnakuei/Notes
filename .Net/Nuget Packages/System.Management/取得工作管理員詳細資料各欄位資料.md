# 取得工作管理員詳細資料各欄位資料

安裝套件
- [System.Management](https://www.nuget.org/packages/System.Management/)

```cs
void Main()
{
	var objects = GetWin32_Process("csrss.exe");
	objects.Dump();

	System.Diagnostics.Process.GetProcessesByName("csrss")
	.SelectMany(p => GetWin32_Process(p) )
	.Dump();
}

public IEnumerable<Win32_Process> GetWin32_Process(Process process)
{
	string query =
	@$"SELECT *
      FROM Win32_Process
      WHERE ProcessId = '{process.Id}'";

	using var searcher = new ManagementObjectSearcher(query);
	var collection = searcher.Get();
	var objects = collection.OfType<ManagementObject>();
	// objects.Dump();
	var result = objects.Select(o => ToWin32_Process(o));
	return result;
}

public IEnumerable<Win32_Process> GetWin32_Process(string processName)
{
	string query =
	@$"SELECT *
      FROM Win32_Process
      WHERE Name = '{processName}'";

	using var searcher = new ManagementObjectSearcher(query);
	var collection = searcher.Get();
	var objects = collection.OfType<ManagementObject>();
	// objects.Dump();
	var result = objects.Select(o => ToWin32_Process(o));
	return result;
}

public Win32_Process ToWin32_Process(ManagementObject o)
{
	var propertyNames = typeof(Win32_Process).GetProperties().Select(p => p.Name).ToArray();

	var result = new Win32_Process();
	foreach (var p in propertyNames)
	{
		var value = o[p];
		if (value != null)
		{
			var property = result.GetType().GetProperty(p);
			try
			{
				property.SetValue(result, value);
			}
			catch (Exception e)
			{
				p.Dump();
				e.Dump();
			}
		}
	}

	return result;
}
```
