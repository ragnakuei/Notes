# 加入事件

```js
const userId : FormlyFieldConfig = {
    key: 'userId',
    type: 'input',
    templateOptions: {
        label: '使用者編號',
        required: true,
    },
};

userId.templateOptions!.blur = () => this.blurUserId();
```

也可以直接加在宣告上面

```js
const userId : FormlyFieldConfig = {
    key: 'userId',
    type: 'input',
    templateOptions: {
        label: '使用者編號',
        required: true,
        blur = () => this.blurUserId();
    },
};
```
