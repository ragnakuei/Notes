# 圖片載入後，再顯示

```html
<html>
<head>

    <style>
        .img.hide {
            width: 0;
            height: 0;
            opacity: 0;
        }

        .img {
            width: 320px;
            height: 240px;
            transition: width 1s, height 1s, opacity 1s;
        }
    </style>
</head>
<body>

<div id="app">
    <h2>Example 21</h2>
    <div>
        <p v-if="!isImageLoaded">Loading...</p>
        <img src="https://loremflickr.com/320/240"
             alt=""
             v-bind:class="imgClass"
             v-on:load="isImageLoaded = true"/>
    </div>
</div>

<script src="https://unpkg.com/vue@next"></script>
<script>
    const {createApp, ref, computed, reactive, watch, onMounted} = Vue;

    createApp({
        setup() {

            const isImageLoaded = ref(false);

            const imgClass = computed(() => {
                return {
                    img: true,
                    hide: !isImageLoaded.value,
                }
            });

            onMounted(() => {
            });

            return {
                isImageLoaded,
                imgClass,
            };
        },
    }).mount("#app");

</script>
</body>
</html>
```
