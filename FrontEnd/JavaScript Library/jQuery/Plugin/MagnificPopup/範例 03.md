# 範例 03

多個 Popup，缺點：會閃的很不舒服 !


```html
<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/magnific-popup.js/1.1.0/magnific-popup.css" />
<script src="https://cdnjs.cloudflare.com/ajax/libs/jquery/2.1.3/jquery.min.js"></script>
<script src="https://cdnjs.cloudflare.com/ajax/libs/magnific-popup.js/1.1.0/jquery.magnific-popup.min.js"></script>
<style>
    .white-popup {
        position: relative;
        background: #FFF;
        padding: 20px;
        width: auto;
        max-width: 500px;
        margin: 20px auto;
    }
</style>

<div>
    <a href="#" onclick='openPopup1()'>openPopup1</a>
</div>

<div id="popup1" class="white-popup mfp-hide">
    Popup content
    <div>
        <input type="text" />
    </div>
    <div>
        <a href="#" onclick='closePopup1()'>closePopup1</a>
    </div>
    <div>
        <a href="#" onclick='openPopup2()'>openPopup2</a>
    </div>
</div>

<div id="popup2" class="white-popup mfp-hide">
    Popup content
    <input type="text" />
    <div>
        <a href="#" onclick='closePopup2()'>closePopup2</a>
    </div>
</div>

<script>
    function openPopup1() {
        setTimeout(() => {
            $.magnificPopup.open({
                items: {
                    src: '#popup1',
                    type: 'inline',
                }
            });
            window.popup1 = $.magnificPopup.instance;
        }, 0);
    }

    function closePopup1() {
        window.popup1.close();
    }
    
    function openPopup2() {
        closePopup1();
        
        setTimeout(() => {
            $.magnificPopup.open({
                items: {
                    src: '#popup2',
                    type: 'inline',
                }
            });
            window.popup2 = $.magnificPopup.instance;
        }, 0);
    }

    function closePopup2() {
        window.popup2.close();
        openPopup1();
    }
</script>
```
