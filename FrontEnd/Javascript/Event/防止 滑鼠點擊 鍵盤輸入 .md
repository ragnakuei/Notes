# 防止 滑鼠點擊 鍵盤輸入 

```js
window.input_locker = {
    prevent_Action : function (e) {
        e.stopPropagation();
        e.preventDefault();
        e.stopImmediatePropagation();
        return false;
    },
    enable: function () {
        document.addEventListener("keydown", input_locker.prevent_Action);
        
        // 可以防止點進 input 中
        document.addEventListener("mousedown", input_locker.prevent_Action);
        document.addEventListener("mouseup", input_locker.prevent_Action);
        document.addEventListener("click", input_locker.prevent_Action);

        console.log("enable keyborard lock");
    },
    disable: function () {
        document.removeEventListener("keydown", input_locker.prevent_Action);
        
        document.removeEventListener("mousedown", input_locker.prevent_Action);
        document.removeEventListener("mouseup", input_locker.prevent_Action);
        document.removeEventListener("click", input_locker.prevent_Action);

        console.log("disable keyborard lock");
    },
}
```
