# 扇型

[參考資料](https://www.cnblogs.com/wangmeijian/p/15994965.html)

```html
<html lang="en">

<head>
    <meta charset="utf-8">
    <style>
        [v-cloak] {
            display: none;
        }
    </style>
</head>

<body>
    <div id="app" v-cloak>

        <div>
            radius：<input type="range" min="0" max="500" step="1" v-model="radius" /> {{ radius }}
        </div>
        <div>
            angle：<input type="range" min="0" max="360" step="1" v-model="angle" /> {{ angle }}
        </div>

        <div style="color: #37CACA">
            <svg v-bind:width="4 * radius" 
                 v-bind:height="4 * radius" 
                 v-bind:viewBox="viewBox" 
                 fill="none">

                <!-- 內圈 -->
                <circle v-bind:r="radius" 
                        v-bind:cx="2 * radius" 
                        v-bind:cy="2 * radius"  
                        fill="#fccccc" />
                <!-- 外圈 -->
                <circle 
                    v-bind:r="radius" 
                    v-bind:cx="2 * radius" 
                    v-bind:cy="2 * radius" 
                    fill="transparent" 
                    stroke="currentColor" 
                    v-bind:stroke-width="2 * radius"
                    v-bind:transform="transform" 
                    v-bind:stroke-dasharray="stroke_dasharray"
                    v-bind:stroke-dashoffset="stroke_dashoffset" />
            </svg>
        </div>
    </div>

    <script src="https://unpkg.com/vue@next"></script>
    <script>
        const { createApp, ref, computed } = Vue;

        const app = createApp({
            setup() {

                const radius = ref(100);

                const viewBox = computed(() => {
                    return `0 0 ${4 * radius.value} ${4 * radius.value}`;
                });

                const stroke_dasharray = computed(() => {
                    return 2 * Math.PI * radius.value;
                });

                const angle = ref(45);

                // 依照 angle 的值，計算 stroke_dashoffset 的值，來顯示扇型要的角度
                const stroke_dashoffset = computed(() => {
                    return 2 * Math.PI * radius.value * (360 - angle.value) / 360;
                });

                // -90度，因為起始點在右方，轉-90讓起點在上方
                const transform = computed(() => {
                    return `rotate(-90 ${2 * radius.value} ${2 * radius.value})`;
                });

                return {
                    radius,
                    viewBox,
                    stroke_dasharray,
                    angle,
                    stroke_dashoffset,
                    transform,
                }
            }
        });

        const vm = app.mount('#app');
    </script>

</body>

</html>
```
